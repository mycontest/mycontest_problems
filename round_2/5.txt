#include <stdio.h>
 
const mn=10000;
int n,a[mn]={2},i,j,m[mn];
 
int div2(int k){
  int i,x,c=0;
  for(i=k;i>=0;i--){
    x=m[i]+10*c;
    m[i]=x/2;
    c=x-2*m[i];
  }
  return c;
}
 
int main(){
  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);
  scanf("%d", &n);
  for(i=1;i<n;i++){
    a[i]=1;
    for(j=0;j<=i;j++) m[j]=a[j];
    for(j=0;j<=i;j++)
      if(div2(i)){a[i]=2;break;}
  }
  for(i=n-1;i>=0;i--) printf("%d",a[i]);
  return 0;
}
